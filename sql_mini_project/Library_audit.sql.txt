

-- 0. DATABASE SETUP

DROP DATABASE IF EXISTS Library_audit_system;
CREATE DATABASE Library_audit_system;
USE Library_audit_system;

-- 1. TABLE CREATION (DDL)

CREATE TABLE Books (
    BookID INT PRIMARY KEY,
    Title VARCHAR(100) NOT NULL,
    Author VARCHAR(100),
    Category VARCHAR(50),
    AvailableCopies INT
);

CREATE TABLE Students (
    StudentID INT PRIMARY KEY,
    Name VARCHAR(100) NOT NULL,
    Email VARCHAR(100) UNIQUE,
    Department VARCHAR(50),
    JoinDate DATE,
    Status VARCHAR(20) DEFAULT 'Active'
);

CREATE TABLE IssuedBooks (
    IssueID INT PRIMARY KEY,
    BookID INT,
    StudentID INT,
    IssueDate DATE NOT NULL,
    ReturnDate DATE,
    FOREIGN KEY (BookID) REFERENCES Books(BookID),
    FOREIGN KEY (StudentID) REFERENCES Students(StudentID)
);

-- 3. OVERDUE BOOKS QUERY


SELECT 
    s.Name AS StudentName,
    b.Title AS BookTitle,
    i.IssueDate,
    DATEDIFF(CURRENT_DATE, i.IssueDate) AS DaysOverdue
FROM IssuedBooks i
JOIN Students s ON i.StudentID = s.StudentID
JOIN Books b ON i.BookID = b.BookID
WHERE i.ReturnDate IS NULL
AND i.IssueDate < CURRENT_DATE - INTERVAL 14 DAY;


-- 4. PENALTY REPORT (₹5 per day after 14 days)


SELECT 
    s.Name AS StudentName,
    b.Title AS BookTitle,
    (DATEDIFF(CURRENT_DATE, i.IssueDate) - 14) AS OverdueDays,
    (DATEDIFF(CURRENT_DATE, i.IssueDate) - 14) * 5 AS PenaltyAmount
FROM IssuedBooks i
JOIN Students s ON i.StudentID = s.StudentID
JOIN Books b ON i.BookID = b.BookID
WHERE i.ReturnDate IS NULL
AND i.IssueDate < CURRENT_DATE - INTERVAL 14 DAY;


-- 5. BOOK POPULARITY ANALYSIS


SELECT 
    b.Category,
    COUNT(i.IssueID) AS TimesBorrowed
FROM IssuedBooks i
JOIN Books b ON i.BookID = b.BookID
GROUP BY b.Category
ORDER BY TimesBorrowed DESC;


-- 6. MARK INACTIVE STUDENTS (3 YEARS)


SET SQL_SAFE_UPDATES = 0;

UPDATE Students s
LEFT JOIN IssuedBooks i ON s.StudentID = i.StudentID
SET s.Status = 'Inactive'
GROUP BY s.StudentID
HAVING MAX(i.IssueDate) IS NULL
   OR MAX(i.IssueDate) < CURRENT_DATE - INTERVAL 3 YEAR;


-- 7. VIEW FOR OVERDUE REPORT


CREATE VIEW OverdueReport AS
SELECT 
    s.Name AS StudentName,
    b.Title AS BookTitle,
    i.IssueDate,
    DATEDIFF(CURRENT_DATE, i.IssueDate) AS DaysOverdue
FROM IssuedBooks i
JOIN Students s ON i.StudentID = s.StudentID
JOIN Books b ON i.BookID = b.BookID
WHERE i.ReturnDate IS NULL
AND i.IssueDate < CURRENT_DATE - INTERVAL 14 DAY;
